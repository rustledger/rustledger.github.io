Quick Start

# Install
cargo install rustledger-cli

# Validate your ledger
bean-check ledger.beancount

# Format in place
bean-format --in-place ledger.beancount

# Query your data
bean-query ledger.beancount "SELECT account, SUM(position) GROUP BY account"

# Interactive query shell
bean-query ledger.beancount
